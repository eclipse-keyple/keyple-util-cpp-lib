32const std::vector<uint8_t> ByteArrayUtil::extractBytes(const std::vector<uint8_t>& src,
33 const int bitOffset,
34 const int nbBytes)
35{
36 if (bitOffset < 0) {
37 throw ArrayIndexOutOfBoundsException("negative index");
38 }
39
40 if (nbBytes < 0) {
41 throw NegativeArraySizeException("negative array size");
42 }
43
44 const int byteOffset = bitOffset / 8;
45 int lBitOffset = bitOffset % 8;
46
47 if (byteOffset >= static_cast<int>((src.size() - (lBitOffset ? 1 : 0)))) {
48 throw ArrayIndexOutOfBoundsException("pos + offset > src size");
49 }
50
51 std::vector<uint8_t> dest(nbBytes);
52
53 if (lBitOffset == 0) {
54 System::arraycopy(src, byteOffset, dest, 0, nbBytes);
55 } else {
56 const int rightShift = 8 - bitOffset;
57 for (int i = 0, j = byteOffset; j < byteOffset + nbBytes; i++, j++) {
58 dest[i] = ((src[j] << lBitOffset) | ((src[j + 1] & 0xFF) >> rightShift));
59 }
60 }
61
62 return dest;
63}

65const std::vector<uint8_t> ByteArrayUtil::extractBytes(const uint64_t src, const int nbBytes)
66{
67 if (nbBytes < 0) {
68 throw NegativeArraySizeException("negative array size");
69 }
70
71 std::vector<uint8_t> data(nbBytes);
72 int shift = 0;
73 int i = nbBytes - 1;
74
75 while (i >= 0) {
76 data[i] = ((src >> shift) & 0xFF);
77 shift += 8;
78 i--;
79 }
80
81 return data;
82}

84uint16_t ByteArrayUtil::extractShort(const std::vector<uint8_t>& src, const int offset)
85{
86 if (offset < 0 || offset > static_cast<int>(src.size() - 2)) {
87 throw ArrayIndexOutOfBoundsException("offset not in range [0...(src.size() - 2)");
88 }
89
90 return ((src[offset] & 0xFF) << 8) | (src[offset + 1] & 0xFF);

93uint32_t ByteArrayUtil::extractInt(const std::vector<uint8_t>& src,
94 const int offset,
95 const int nbBytes,
96 const bool isSigned)
97{
98 if (offset < 0) {
99 throw ArrayIndexOutOfBoundsException("negative offset");
100 }
101
102 if ((offset + nbBytes) > static_cast<int>(src.size())) {
103 throw ArrayIndexOutOfBoundsException("offset + nbBytes > src size");
104 }
105
106 /* C++ - doesn't make sense to have nbBytes > int size */
107 if (nbBytes > 4) {
108 throw IllegalArgumentException("nbBytes can't be bigger than 4");
109 }
110
111 uint32_t val = 0;
112 uint32_t complement = 0xFFFFFFFF;
113 int lOffset = offset;
114 int lNbBytes = nbBytes;
115 bool negative = false;
116
117 /* Get value */
118 while (lNbBytes > 0) {
119 /* Check MSB byte negativeness */
120 negative = negative ? true : ((src[lOffset] & 0xFF) > 0x7F ? true : false);
121 val |= ((src[lOffset++] & 0xFF) << (8 * (--lNbBytes)));
122 complement &= ~(0xFF << 8 * lNbBytes);
123 }
124
125 /* If signed, add complement */
126 if (isSigned && negative) {
127 val |= complement;
128 }
129
130 return (uint32_t)val;
131}

133uint64_t ByteArrayUtil::extractLong(const std::vector<uint8_t>& src,
134 const int offset,
135 const int nbBytes,
136 const bool isSigned)
137{
138 int lOffset = offset;
139 int lNbBytes = nbBytes;
140
141
142 if (lOffset < 0 || lOffset > static_cast<int>(src.size() - lNbBytes)) {
143 throw ArrayIndexOutOfBoundsException("offset not in range [0...(src.size() - nbBytes)");
144 }
145
146 uint64_t val = 0L;
147 uint64_t complement = 0xFFFFFFFFFFFFFFFF;
148 bool negative = false;
149
150 /* Get value */
151 while (lNbBytes > 0) {
152 /* Check MSB byte negativeness */
153 negative = negative ? true : ((src[lOffset] & 0xFF) > 0x7F ? true : false);
154 val |= (((uint64_t)(src[lOffset++] & 0xFF)) << (8 * (--lNbBytes)));
155 complement &= ~(((uint64_t)0xFF) << 8 * lNbBytes);
156 }
157
158 /* If signed, add complement */
159 if (isSigned && negative) {
160 val |= complement;
161 }
162
163 return (uint64_t)val;
164}
